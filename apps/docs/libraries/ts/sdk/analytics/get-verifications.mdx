---
title: "Get Verifications"
description: "Get verification analytics for a project or customer"
---

<RequestExample>

```ts
const { result, error } = await unprice.analytics.getVerifications({
  projectId: "project_1234567890",
  range: "24h",
  customerId: "cus_1234567890" // optional
});

if (error) {
  // handle potential network or bad request error
  // a link to our docs will be in the `error.docs` field
  console.error(error.message);
  return;
}

console.log(result);
```

</RequestExample>

<ResponseExample>
```json
{
  "verifications": [
    {
      "projectId": "project_1234567890",
      "customerId": "cus_1234567890",
      "featureSlug": "feature_1",
      "count": 100,
      "success": 90,
      "failure": 10
    }
  ]
}
```
</ResponseExample>

## Request

<ParamField body="projectId" type="string" required>
    The ID of the project to get verifications for
</ParamField>

<ParamField body="range" type="string" enum="60m,24h,7d,30d,90d" required>
    The time range to get verifications for (last hour, day, week, month, or 3 months)
</ParamField>

<ParamField body="customerId" type="string">
    The ID of the customer to get verifications for (optional)
</ParamField>

## Response

<ResponseField name="verifications" type="array" required>
    Array of verification data points
</ResponseField>

<ResponseField name="verifications[].projectId" type="string" required>
    The ID of the project
</ResponseField>

<ResponseField name="verifications[].customerId" type="string">
    The ID of the customer
</ResponseField>

<ResponseField name="verifications[].featureSlug" type="string" required>
    The slug of the feature
</ResponseField>

<ResponseField name="verifications[].count" type="number" required>
    The total number of verifications
</ResponseField>

<ResponseField name="verifications[].success" type="number" required>
    The number of successful verifications
</ResponseField>

<ResponseField name="verifications[].failure" type="number" required>
    The number of failed verifications
</ResponseField>
