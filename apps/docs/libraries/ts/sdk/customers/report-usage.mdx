---
title: "Report Usage"
description: "Report usage for a customer"
---

<RequestExample>

```ts
const { result, error } = await unprice.customers.reportUsage({
  customerId: "cus_1234567890",
  featureSlug: "tokens",
  usage: 30,
  idempotenceKey: "123e4567-e89b-12d3-a456-426614174000",
  metadata: {
    action: "create",
    country: "US"
  }
});

if (error) {
  // handle potential network or bad request error
  // a link to our docs will be in the `error.docs` field
  console.error(error.message);
  return;
}

console.log(result);
```

</RequestExample>

<ResponseExample>
```json
{
  "success": true,
  "message": "Usage reported successfully",
  "cacheHit": false,
  "remaining": 970
}
```
</ResponseExample>

## Request

<ParamField body="customerId" type="string" required>
    The ID of the customer to report usage for
</ParamField>

<ParamField body="featureSlug" type="string" required>
    The slug of the feature to report usage for
</ParamField>

<ParamField body="usage" type="number" required>
    The amount of usage to report
</ParamField>

<ParamField body="idempotenceKey" type="string" format="uuid" required>
    A unique key to prevent duplicate usage reports
</ParamField>

<ParamField body="metadata" type="object">
    Additional metadata about the usage report
</ParamField>

## Response

<ResponseField name="success" type="boolean" required>
    Whether the usage was reported successfully
</ResponseField>

<ResponseField name="message" type="string">
    A message describing the result of the operation
</ResponseField>

<ResponseField name="cacheHit" type="boolean">
    Whether the request was served from cache
</ResponseField>

<ResponseField name="remaining" type="number">
    The remaining usage limit for the feature
</ResponseField>
