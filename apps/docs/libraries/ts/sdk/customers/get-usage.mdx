---
title: "Get Usage"
description: "Get usage for a customer"
---

<RequestExample>

```ts
const { result, error } = await unprice.customers.getUsage("cus_1234567890");

if (error) {
  // handle potential network or bad request error
  // a link to our docs will be in the `error.docs` field
  console.error(error.message);
  return;
}

console.log(result);
```

</RequestExample>

<ResponseExample>
```json
{
  "usage": [
    {
      "featureSlug": "feature_1",
      "used": 100,
      "limit": 1000,
      "resetAt": 1681492000
    }
  ]
}
```
</ResponseExample>

## Request

<ParamField path="customerId" type="string" required>
    The ID of the customer to get usage for
</ParamField>

## Response

<ResponseField name="usage" type="array" required>
    Array of usage information for each feature
</ResponseField>

<ResponseField name="usage[].featureSlug" type="string" required>
    The slug of the feature
</ResponseField>

<ResponseField name="usage[].used" type="number" required>
    The amount of usage consumed
</ResponseField>

<ResponseField name="usage[].limit" type="number" required>
    The usage limit for the feature
</ResponseField>

<ResponseField name="usage[].resetAt" type="number" required>
    The timestamp when the usage will reset
</ResponseField>
