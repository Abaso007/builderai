---
title: "Get Entitlements"
description: "Get entitlements for a customer"
---

<RequestExample>

```ts
const { result, error } = await unprice.customers.getEntitlements("cus_1234567890");

if (error) {
  // handle potential network or bad request error
  // a link to our docs will be in the `error.docs` field
  console.error(error.message);
  return;
}

console.log(result);
```

</RequestExample>

<ResponseExample>
```json
{
  "entitlements": [
    {
      "featureSlug": "feature_1",
      "granted": true,
      "usage": {
        "used": 100,
        "limit": 1000
      }
    }
  ]
}
```
</ResponseExample>

## Request

<ParamField path="customerId" type="string" required>
    The ID of the customer to get entitlements for
</ParamField>

## Response

<ResponseField name="entitlements" type="array" required>
    Array of entitlements for the customer
</ResponseField>

<ResponseField name="entitlements[].featureSlug" type="string" required>
    The slug of the feature
</ResponseField>

<ResponseField name="entitlements[].granted" type="boolean" required>
    Whether the feature is granted to the customer
</ResponseField>

<ResponseField name="entitlements[].usage" type="object">
    Usage information for the feature
</ResponseField>

<ResponseField name="entitlements[].usage.used" type="number">
    The amount of usage consumed
</ResponseField>

<ResponseField name="entitlements[].usage.limit" type="number">
    The usage limit for the feature
</ResponseField>
