---
title: "Get Active Phase"
description: "Get active phase for a customer"
---

<RequestExample>

```ts
const { result, error } = await unprice.customers.getActivePhase("cus_1234567890");

if (error) {
  // handle potential network or bad request error
  // a link to our docs will be in the `error.docs` field
  console.error(error.message);
  return;
}

console.log(result);
```

</RequestExample>

<ResponseExample>
```json
{
  "phase": {
    "id": "phase_1234567890",
    "startDate": 1678900000,
    "endDate": 1681492000,
    "planVersionId": "pv_1234567890",
    "status": "active"
  }
}
```
</ResponseExample>

## Request

<ParamField path="customerId" type="string" required>
    The ID of the customer to get active phase for
</ParamField>

## Response

<ResponseField name="phase" type="object" required>
    The active phase information
</ResponseField>

<ResponseField name="phase.id" type="string" required>
    The ID of the phase
</ResponseField>

<ResponseField name="phase.startDate" type="number" required>
    The start timestamp of the phase
</ResponseField>

<ResponseField name="phase.endDate" type="number" required>
    The end timestamp of the phase
</ResponseField>

<ResponseField name="phase.planVersionId" type="string" required>
    The ID of the plan version for this phase
</ResponseField>

<ResponseField name="phase.status" type="string" required>
    The status of the phase (active, upcoming, etc.)
</ResponseField>
