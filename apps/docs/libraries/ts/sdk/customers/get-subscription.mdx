---
title: "Get Subscription"
description: "Get subscription with the active phase for a customer"
---

<RequestExample>

```ts
const { result, error } = await unprice.customers.getSubscription("cus_1234567890");

if (error) {
  // handle potential network or bad request error
  // a link to our docs will be in the `error.docs` field
  console.error(error.message);
  return;
}

console.log(result);
```

</RequestExample>

<ResponseExample>
```json
{
  "subscription": {
    "id": "sub_1234567890",
    "status": "active",
    "currentPeriodStart": 1678900000,
    "currentPeriodEnd": 1681492000,
    "cancelAtPeriodEnd": false,
    "phases": [
      {
        "id": "phase_1234567890",
        "startDate": 1678900000,
        "endDate": 1681492000,
        "planVersionId": "pv_1234567890",
        "status": "active"
      }
    ]
  }
}
```
</ResponseExample>

## Request

<ParamField path="customerId" type="string" required>
    The ID of the customer to get subscription for
</ParamField>

## Response

<ResponseField name="subscription" type="object" required>
    The subscription information
</ResponseField>

<ResponseField name="subscription.id" type="string" required>
    The ID of the subscription
</ResponseField>

<ResponseField name="subscription.status" type="string" required>
    The status of the subscription (active, canceled, etc.)
</ResponseField>

<ResponseField name="subscription.currentPeriodStart" type="number" required>
    The start timestamp of the current period
</ResponseField>

<ResponseField name="subscription.currentPeriodEnd" type="number" required>
    The end timestamp of the current period
</ResponseField>

<ResponseField name="subscription.cancelAtPeriodEnd" type="boolean" required>
    Whether the subscription will be canceled at the end of the current period
</ResponseField>

<ResponseField name="subscription.phases" type="array" required>
    Array of subscription phases
</ResponseField>

<ResponseField name="subscription.phases[].id" type="string" required>
    The ID of the phase
</ResponseField>

<ResponseField name="subscription.phases[].startDate" type="number" required>
    The start timestamp of the phase
</ResponseField>

<ResponseField name="subscription.phases[].endDate" type="number" required>
    The end timestamp of the phase
</ResponseField>

<ResponseField name="subscription.phases[].planVersionId" type="string" required>
    The ID of the plan version for this phase
</ResponseField>

<ResponseField name="subscription.phases[].status" type="string" required>
    The status of the phase (active, upcoming, etc.)
</ResponseField>
